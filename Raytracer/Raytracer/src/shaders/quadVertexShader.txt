#version 130

/* position of vertex as 2d vector */
in vec2 vertex;

/* Interpolated texture coordinate to fragment shader */
out vec2 texcoord;

void main(void)
{
	gl_Position = vec4(vertex, 0.0, 1.0);

	/*
	*	Compute texture co-ordinate by simply
	*	interval-mapping from [-1..+1] to [0..1]
	*/
	texcoord = vertex * 0.5 + vec2(0.5, 0.5);
}
